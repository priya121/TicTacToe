package ttt;

public class ComputerPlayer {
    IO io;
    Board board;
    int number;
    Symbol symbol;

    public ComputerPlayer(Board board) {
    this.board = board;
    this.symbol = Symbol.NOUGHT;
    }

    public int numberGenerated() {
    RandomNumber random = new RandomNumber();
    number = random.generate(8);
        while (!board.isValid(number)) {
            number = random.generate(8);
        }
    return number;
    }
}
