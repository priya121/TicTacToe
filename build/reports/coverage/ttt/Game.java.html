package ttt;

import java.util.Arrays;
import java.util.List;

public class Game {
    private int counter = 0;
    private List<String> emptyBoard;
    private Board board;
    private DisplayBoard display;
    private Player player;
    private String cross;
    private String nought;
    IO io;

    public Game(Board board, IO io) {
        String empty = " ";
        emptyBoard = Arrays.asList(empty, empty, empty,
                                   empty, empty, empty,
                                   empty, empty, empty);
        cross = "X";
        nought = "O";
        this.board = board;
        this.io = io;
        this.player = new Player(io, board);
    }

    public static void main(String[] args) {
        String empty = " ";
        List<String> emptyBoard = Arrays.asList(empty, empty, empty,
                                                empty, empty, empty,
                                                empty, empty, empty);
        Board board = new Board(emptyBoard);
        IO io = new ConsoleIO(System.in, System.out);
        Player player = new Player(io, board);
        Game newGame = new Game(board, io);
        newGame.gameLoop();
    }

    public void gameLoop() {
        while (!checkForWin() && counter < 9) {
            io.showOutput("next player");
            player.markBoard();
            counter += 1;
            DisplayBoard currentDisplay = new DisplayBoard(board.getCurrentBoard());
            io.showOutput(currentDisplay.showBoard());
        }
        io.showOutput("the winner is " + player.getSymbol());
        io.showOutput("game over");
    }

    public boolean checkForWin() {
        if (isWinningRow() || isWinningDiagonal() || isWinningColumn())  {
            return true;
        }
        return false;
    }

    public boolean isWinningRow() {
        if (board.get(0).equals(board.get(1)) && board.get(1).equals(board.get(2))) {
            return true;
        }
        if (board.get(3).equals(board.get(4)) && board.get(4).equals(board.get(5))) {
            return true;
        }
        if (board.get(6).equals(board.get(7)) && board.get(7).equals(board.get(8))) {
            return true;
        }
        return false;
    }

    public boolean isWinningColumn() {
        if (board.get(0).equals(board.get(3)) && board.get(3).equals(board.get(6))) {
            return true;
        }
        if (board.get(1).equals(board.get(4)) && board.get(4).equals(board.get(7))) {
            return true;
        }
        if (board.get(2).equals(board.get(5)) && board.get(5).equals(board.get(8))) {
            return true;
        }
        return false;
    }


    public boolean isWinningDiagonal() {
        if (board.get(0).equals(board.get(4)) && board.get(4).equals(board.get(8))) {
            return true;
        }
        if (board.get(2).equals(board.get(4)) && board.get(4).equals(board.get(6))) {
            return true;
        }
        return false;
    }
}
